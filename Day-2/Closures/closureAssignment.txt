create an object (how?) in javascript and assign it to a variable "spinner".. The object should behave the following:

spinner.up() => 1
spinner.up() => 2
spinner.up() => 3
spinner.up() => 4

spinner.down() => 3
spinner.down() => 2
spinner.down() => 1
spinner.down() => 0
spinner.down() => -1

solution
=========
function getSpinner(){
  var counter = 0;
  function increment(){
     return ++counter;
  }
  function decrement(){
     return --counter;
  }
  var obj =  {
     up : increment,
     down : decrement
  };
  return obj;
}
